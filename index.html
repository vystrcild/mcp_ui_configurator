Installation

pip install "mcp[cli]"

Example Usage

import mcp
from mcp.client.streamable_http import streamablehttp_client

async def main():
    url = "https://mcp.apify.com/"
    async with streamablehttp_client(url) as (read_stream, write_stream):
        async with mcp.ClientSession(read_stream, write_stream) as session:
            await session.initialize()
            tools_result = await session.list_tools()
            print("Available tools:", [t.name for t in tools_result.tools])

Installation

npm install @modelcontextprotocol/sdk

Example Usage

import { Client } from "@modelcontextprotocol/sdk/client/index.js";
import { StreamableHTTPClientTransport } from "@modelcontextprotocol/sdk";

const serverUrl = "https://mcp.apify.com/";
const transport = new StreamableHTTPClientTransport(serverUrl);

const client = new Client({
    name: "My App",
    version: "1.0.0"
});

await client.connect(transport);
const tools = await client.listTools();
console.log("Available tools:", tools.map(t => t.name));

JSON Configuration for Mac/Linux

{
  "mcpServers": {
    "apify": {

        "-y",
        "@apify/mcp-server",
        "run",
        "@apify/mcp-server",
        "--profile",
        "default"
      ]
    }
  }
}
